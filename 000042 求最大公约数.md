# The Greatest Common Divisor

## Code

```C
#include <stdio.h>

int main(void)
{
	int num1;
	int num2;
	int max = 1;

	printf("请输入数字1。\n");
	scanf_s("%d", &num1);

	printf("请输入数字2。\n");
	scanf_s("%d", &num2);

	for (int a = 2; a <= num1; a++) {
		for (int b = 2; b <= num2; b++) {
			if ( ( ( num1 % a ) == 0 ) && ( ( num2 % b ) == 0 ) && ( a == b ) ) {
				if (a > max) {
					max = a;
				}
			}
		}
	}

	printf("这俩个数的最大公约数是：%d。\n", max);

	return 0;
}
```
