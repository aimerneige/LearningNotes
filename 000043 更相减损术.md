# The Greatest Common Divisor

## Code

```C
#include <stdio.h>

int main(void)
{
	int num1;
	int num2;

	printf("请输入第一个数字。\n");
	scanf_s("%d", &num1);

	printf("请输入第二个数字。\n");
	scanf_s("%d", &num2);

	do {
		if (num1 > num2) {
			num1 -= num2;
		}
		else {
			num2 -= num1;
		}
	} while (num1 != num2);

	printf("这俩个数的最大公约数是：%d。\n", num1);

	return 0;
}
```
