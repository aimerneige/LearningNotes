# Elephant Drink Water

## Code

```C
#include <stdio.h>

int main(void)
{
    const double PI = 3.1415926535;
    const int water = 20;
    int r, h;
    double V, ret;
    scanf("%d %d", &r, &h);
    V = PI * r * r * h;
    ret = (water * 1000) / V;
    int temp;
    temp = ret;
    if (temp < ret) temp++;
    printf("%d", temp);

    return 0;
}
```
