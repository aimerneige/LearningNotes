# Count

## Code

```C
#include <stdio.h>

int main(void)
{
	int number[100];
	int num, cnt = 0, temp, x = 0;
	// 初始化

	printf("你只能输入[0,9]范围内的整数，否则会导致程序崩溃，输入-1表示结束。\n");
	// 提示

	scanf("%d", &num);
	while (num != -1){
		number[cnt] = num;
		cnt++;
		scanf("%d", &num);
	}
	// 读入数据

	for(int n = 0; n < 10; n++) {
		for (int i = 0; i < cnt; i++) {
			temp = number[i];
			if (temp == n) {
				x++;
			}
		}
		printf("程序中出现了%d个%d。\n", x, n);
		x = 0;// 重置计数器
	}
	// 计算

	return 0;
}
```
