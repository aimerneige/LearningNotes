# Count-Classics

## Code1

```C
#include <stdio.h>

int main()
{
	int count[10];
	int x;
	// 声明

	for (int i = 0; i < 10; i++) {
		count[i] = 0;
	}
	// 初始化数组

	scanf("%d", &x);
	while (x != -1) {
		if (x >= 0 && x <= 9) {
			count[x]++;
		} else {
			printf("Wrong numbers!\n");
		}
		scanf("%d", &x);
	}
	// 数据的读取

	for (int i = 0; i < 10; i++)
	{
		printf("%d: %d\n", i, count[i]);
	}
	// 数据读取

	return 0;
}
```
