struct date {
    int year;
    int month;
    int day;
};
int LeapYear(int year);
struct date DaysAfter(struct date sdate, int days);
const int monthday[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
const int yearday[2] = {365, 366};
int main(void)
{
    struct date start = {1777, 4, 30};
    struct date end;
    int days;
    days = 8113;
    end = DaysAfter(start, days);
    printf("%i-%i-%i", end.year, end.month, end.day);
    return 0;
}
// 输入年份，若是闰年返回1，否则返回0。
int LeapYear(int year)
{
    int ret = 0;
    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0 && year % 3200 != 0)) ret = 1;
    return ret;
}
// 输入日期和天数x，返回经过x天后的日期
struct date DaysAfter(struct date sdate, int days)
{
    // struct date end;
    int delta;
    delta = monthday[sdate.month - 1] - sdate.day + 1;
    if (LeapYear(sdate.year) && sdate.month == 2) delta++;
    while (days) {
        if (delta < days) {
            days -= delta;
            if (sdate.month < 12) {
                sdate.month++;
                sdate.day = 1;
                for (int i = sdate.month; i <= 12; i++) {
                    int temp = monthday[sdate.month - 1];
                    if (sdate.month == 2) temp++;
                    if (days > temp) {
                        days -= temp;
                        sdate.month++;
                    } else {
                        sdate.day += days;
                        days = 0;
                        break;
                    }
                }
                if (days == 0) break;
                sdate.year++;
                sdate.month = 1;
            } else {
                sdate.year++;
                sdate.month = 1;
                sdate.day = 1;
            }
            while (days > yearday[ LeapYear(sdate.year) ]) {
                sdate.year++;
                days -= yearday[ LeapYear(sdate.year) ];
            }
            while (days > monthday[sdate.month - 1]) {
                if (sdate.month == 12) {
                    sdate.month = 1;
                } else {
                    sdate.month++;
                }
                days -= monthday[sdate.month - 1];
            }
            while (days) {
                sdate.day += days;
                days = 0;
            }
        } else {
            sdate.day += days;
            days = 0;
        }
    }
    if (sdate.day > 1) {
        sdate.day--;
    } else {
        sdate.month--;
        if (LeapYear(sdate.year) && (sdate.month == 2) ) {
            sdate.day = 29;
        } else {
            sdate.day = monthday[sdate.month];
        }
    }
    return sdate;
}

## Exercise

> 题目标题: 高斯日记
>
> 大数学家高斯有个好习惯：无论如何都要记日记。
>
> 他的日记有个与众不同的地方，他从不注明年月日，而是用一个整数代替，比如：4210
>
> 后来人们知道，那个整数就是日期，它表示那一天是高斯出生后的第几天。这或许也是个好习惯，它时时刻刻提醒着主人：日子又过去一天，还有多少时光可以用于浪费呢？
>
> 高斯出生于：1777年4月30日。
>
> 在高斯发现的一个重要定理的日记上标注着：5343，因此可算出那天是：1791年12月15日。
>
> 高斯获得博士学位的那天日记上标着：8113
>
> 请你算出高斯获得博士学位的年月日。
>
> 提交答案的格式是：yyyy-mm-dd, 例如：1980-03-21
>
> 请严格按照格式，通过浏览器提交答案。
>
> 注意：只提交这个日期，不要写其它附加内容，比如：说明性的文字。

bool isLeapYear(int y){
	if((y % 4 == 0 && y % 100 != 0) || (y % 400 == 0))
		return true;
	else return false;
}

int main(){
	int y = 1777;
	int m = 4;
	int d = 30;
	for(int i = 0; i < 8112; i++){

		if((m == 1 || m == 3 || m == 5 || m == 7 || m == 8 || m == 10) && (d == 32)){
			m++;
			d = 1;
		}
		if((m == 4 || m == 6  || m == 9 || m == 11) && d == 31){
			m++;
			d = 1;
		}
		if(m == 12 && d == 32){
			y++;
			m = 1;
			d = 1;
		}
		if(m == 2 && isLeapYear(y) && d == 30){
			m++;
			d = 1;
		}
		if(m == 2 && !isLeapYear(y) && d == 29){
			m++;
			d = 1;
		}
		d++;
	}
	cout << y <<"-" << m << "-" << d<< endl;
	
	return 0;
}
