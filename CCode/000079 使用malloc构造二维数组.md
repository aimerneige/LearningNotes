# Malloc-2D

## Code

```C
#include <stdio.h>
#include <stdlib.h>

int main(void)
{
    int **a;
    int x; // row
    int y; // column
//    scanf("%d", &x);
//    scanf("%d", &y);
    x = 4;
    y = 2;
    a = (int**)malloc( x  * sizeof(int*) );
    for (int i = 0; i < x; i++) {
        a[i] = (int*)malloc( y * sizeof(int) );
    }

    for (int i = 0; i < x; i++) {
        for (int j = 0; j < y; j++) {
            a[i][j] = 5;
        }
    }

    for (int i = 0; i < x; i++) {
        for (int j = 0; j < y; j++) {
            printf("a[%d][%d] = %d\t", i, j, a[i][j]);
        }
        printf("\n");
    }

    { // For Free
        for (int i = 0; i < x; i++) {
            free(a[i]);
        }
        free(a);
    }

    return 0;
}
```

```
a[2][0] = 5     a[2][1] = 5
a[3][0] = 5     a[3][1] = 5
```
